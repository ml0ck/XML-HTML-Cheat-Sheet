XML, HTML, Cheat Sheet

1. Structure Your HTML

Start by creating a basic HTML layout for your webpage.

<!DOCTYPE html>
<html>
<head>
  <title>My Dynamic Website</title>
</head>
<body>
  <h1>Welcome to My Website</h1>
</body>
</html>

2. Add External CSS

Link a CSS file to style your HTML content.

<link rel="stylesheet" href="style.css">

3. Create a Basic XML File

XML files are used to store and transport data. They can be read by JavaScript to update the webpage.

<content>
  <text>This is dynamic content!</text>
</content>

4. Link External JavaScript

Use JavaScript to add interactivity to your webpage.

<script src="script.js"></script>

5. Modify Content Using JavaScript

With JavaScript, you can dynamically update the content of an element.

function changeContent() {
  document.getElementById("content").innerText = "New Dynamic Content";
}

6. Fetch Data from XML

Use JavaScript to retrieve and manipulate data from an XML file.

function fetchXML() {
  let xhr = new XMLHttpRequest();
  xhr.open("GET", "data.xml", true);
  xhr.onload = function() {
    let xmlDoc = xhr.responseXML;
    let text = xmlDoc.getElementsByTagName("text")[0].childNodes[0].nodeValue;
    document.getElementById("content").innerText = text;
  };
  xhr.send();
}

7. Modify Table Content

Update the content inside an HTML table dynamically.

function changeTableContent() {
  document.getElementById("cell1").innerText = "Updated Table Content";
}

8. Dynamically Add an Image

Use JavaScript to add an image to your webpage.

function addImage() {
  let img = document.createElement("img");
  img.src = "new-image.jpg";
  document.body.appendChild(img);
}

9. Use AJAX for Asynchronous Data

AJAX helps in loading data asynchronously without reloading the page.

function fetchData() {
  let xhr = new XMLHttpRequest();
  xhr.open("GET", "data.json", true);
  xhr.onreadystatechange = function() {
    if (xhr.readyState == 4 && xhr.status == 200) {
      let data = JSON.parse(xhr.responseText);
      document.getElementById("content").innerText = data.text;
    }
  };
  xhr.send();
}

10. Form Validation

Ensure the form fields are filled correctly before submitting.

function validateForm() {
  let x = document.forms["myForm"]["fname"].value;
  if (x == "") {
    alert("Name must be filled out");
    return false;
  }
}
